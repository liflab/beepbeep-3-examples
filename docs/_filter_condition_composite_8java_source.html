38 /**
39  * Filter a trace by evaluating a compound condition on the events of
40  * that trace.
41  * <p>
42  * In this example, we wish to keep all events that are even <em>and</em>
43  * greater than 4, and discard the others.
44  * This example can be represented grahpically as:
45  * <p>
46  * <img src="./doc-files/basic/FilterConditionComposite.png" alt="Processor graph">
47  * @author Sylvain Hallé
48  * @difficulty Easy
49  */
50 public class FilterConditionComposite
51 {
52  public static void main(String[] args) throws ConnectorException
53  {
54  // Create a trace of dummy values
55  QueueSource source_values = new QueueSource();
56  source_values.setEvents(new Integer[]{6, 5, 3, 8, 9, 2, 1, 7, 4, 5,
57  2, 4, 7, 6, 12, 8, 1});
58  // Fork the trace in two
59  Fork fork = new Fork(2);
60  connect(source_values, fork);
61  // Connect the first ("left") output of the fork into a filter
62  Filter filter = new Filter();
63  connect(fork, LEFT, filter, LEFT);
64  // Create the compound condition "is even and is greater than 4"
65  FunctionTree tree = new FunctionTree(And.instance,
66  IsEven.instance,
67  new FunctionTree(IsGreaterThan.instance,
68  new ArgumentPlaceholder(0),
69  new Constant(4)));
70  FunctionProcessor condition = new FunctionProcessor(tree);
71  // Connect its input to the second output of the fork
72  connect(fork, RIGHT, condition, INPUT);
73  // Connect the condition as the second input of our filter
74  connect(condition, OUTPUT, filter, RIGHT);
75  // Get a reference to the filter's output pullable
76  Pullable p = filter.getPullableOutput();
77  // Pull 4 events from p
78  for (int i = 0; i < 4; i++)
79  {
80  int x = (Integer) p.pull();
81  System.out.printf("Output event #%d is %d\n", i, x);
82  }
83  }
84 }
