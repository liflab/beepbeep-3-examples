18 package network.httppush;
19 
20 import ca.uqac.lif.cep.Connector;
21 import ca.uqac.lif.cep.ProcessorException;
22 import ca.uqac.lif.cep.Pushable;
23 import ca.uqac.lif.cep.http.HttpDownstreamGateway;
24 import ca.uqac.lif.cep.http.HttpUpstreamGateway;
25 import ca.uqac.lif.cep.io.Print;
26 import ca.uqac.lif.jerrydog.RequestCallback.Method;
27 
28 /**
29  * In this example, Machine A and Machine B are actually the same host; they
30  * just listen to different TCP ports on the same computer. Read this program
31  * first before looking that the other files in this package.
32  * <p>
33  * <img
34  * src="./doc-files/network/httppush/PushLocalSimple.png"
35  * alt="Processor graph">
36  * @author Sylvain Hallé
37  *
38  */
39 public class PushLocalSimple
40 {
41  public static void main(String[] args) throws ProcessorException, InterruptedException
42  {
43  ///
44  HttpUpstreamGateway up_gateway = new HttpUpstreamGateway("http://localhost:12144/push");
45 
46  /* We now move on to Machine B, which is responsible for receiving character
47  * strings and converting them back into objects. This is the mirror process of
48  * what was just done. So, the first step is to create an
49  * {@link HttpDownstreamGateway}. The gateway is instructed to listen to incoming
50  * requests on port 12144, to respond to requests made at the page "/push", and
51  * send through an HTTP <code>POST</code> request. */
52  HttpDownstreamGateway dn_gateway = new HttpDownstreamGateway(12144, "/push", Method.POST);
53 
54  /* Just so that we can see something, we plug a {@link Print} processor at
55  * the end; it will print to the standard output whatever object it receives
56  * from upstream. */
57  Print print = new Print();
58  Connector.connect(dn_gateway, print);
59 
60  /* Since the gateways are actually mini-web servers, the servers need to
61  * be launched so that they can actually communicate. This is done by
62  * calling the {@link Processor#start() start()} method on both
63  * processors. Look out, as there can be only one server on a machine
64  * listening to a given TCP port; if you have another instance of this
65  * program already running, the call to start will throw an Exception. */
66  up_gateway.start();
67  dn_gateway.start();
68 
69  /* We are now ready to push events and see what happens. First, we get
70  * a handle on the {@link Pushable} of the very first processor of the
71  * chain, <code>serialize</code> (which resides on Machine A). */
72  Pushable p = up_gateway.getPushableInput();
73 
74  /* Let's push some dummy object. After the call to push, the standard
75  * output should display the contents of that object. */
76  p.push("foo");
77 
78  /* Sleep a little so you have time to look at the console... */
79  Thread.sleep(1000);
80 
81  /* Let's push again. You know the drill. */
82  p.push("bar");
83 
84  /* Once everything is done, we have to stop the servers to free the
85  * TCP ports on your machine. This is done by calling
86  * {@link Processor#stop() stop()} on both processors. */
87  up_gateway.stop();
88  dn_gateway.stop();
89 
90  /* That's all folks! */
91  ///
92  }
93 }
