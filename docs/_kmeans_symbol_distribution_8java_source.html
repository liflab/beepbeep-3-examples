46 /**
47  * Create clusters over the distribution of symbols in a set of input
48  * streams.
49  * <p>
50  * In this example, input traces are made of symbols <tt>a</tt> and
51  * <tt>b</tt>. A set of seven such traces is located in the file
52  * <tt>strings-1.csv</tt>. For each of these traces, the pattern
53  * processor &beta; computes a <em>feature vector</em> made of two numbers,
54  * corresponding to the fraction of <tt>a</tt>'s and <tt>b</tt>'s in
55  * the trace. This is done by
56  * <ul>
57  * <li>slicing the input trace according to the current symbol</li>
58  * <li>counting the events in each slice</li>
59  * <li>taking the values of the resulting map into a list, and normalizing
60  * them (so that their sum is equal to 1)</li>
61  * </ul>
62  * For example, on the input sequence
63  * <pre>
64  * a, b, a, a, b, b
65  * </pre>
66  * the resulting feature vector would be (0.4, 0.6).
67  * <p>
68  * We then use the K-means clustering algorithm to find the centroids of
69  * two clusters based on those feature vectors.
70  * <p>
71  * The processor mining function is therefore parameterized as follows:
72  * <p>
73  * <table>
74  * <tr><th>Parameter</th><th>Value</th></tr>
75  * <tr>
76  * <td><img src="./doc-files/mining/trenddistance/BetaProcessor.png" alt="Processor graph"></td>
77  * <td><img src="./doc-files/mining/extraction/SymbolDistributionDoublePoint.png" alt="Processor graph"></td>
78  * </tr>
79  * <tr>
80  * <td><img src="./doc-files/mining/extraction/AlphaProcessor.png" alt="Processor graph"></td>
81  * <td><img src="./doc-files/mining/extraction/Kmeans-Function.png" alt="Processor graph"> (K-means with K=2)</td>
82  * </tr>
83  * </table>
84  * <p>
85  * The traces in the input CSV file either have an approximate 30%-70%
86  * distribution of <tt>a</tt>'s and <tt>b</tt>'s, or the reverse. The
87  * feature vectors can be plotted as follows, with each dot representing
88  * the a-b distribution of a single trace.
89  * <p>
90  * <img src="./doc-files/mining/extraction/ClusteringAB.png" alt="Clustering graph">
91  * <p>
92  * Applying the k-means algorithm, with k=2, will compute two cluster centers,
93  * represented by crosses in the above plot.
94  *
95  * @author Sylvain Hall√©
96  *
97  */
98 public class KmeansSymbolDistribution
99 {
100  public static void main(String[] args) throws FunctionException
101  {
102  /* First, we must get from somewhere a set of sequences. For the sake
103  * of this example, we just create a few dummy sequences of numbers
104  * from the contents of a file. */
105  Set<Sequence<String>> sequences = SequenceReader.readStringSequences("strings-1.csv");
106 
107  /* We then create a processor that computes the feature vector
108  * from an input trace. */
109  GroupProcessor vector = new GroupProcessor(1, 1);
110  {
111  GroupProcessor counter = new GroupProcessor(1, 1);
112  {
113  ReplaceWith one = new ReplaceWith(new Constant(1));
114  counter.associateInput(INPUT, one, INPUT);
115  Cumulate sum_one = new Cumulate(new CumulativeFunction<Number>(Numbers.addition));
116  Connector.connect(one, sum_one);
117  counter.associateOutput(OUTPUT, sum_one, OUTPUT);
118  counter.addProcessors(one, sum_one);
119  }
120  Slice slicer = new Slice(new IdentityFunction(1), counter);
121  ApplyFunction to_normalized_vector = new ApplyFunction(
122  new FunctionTree(Normalize.instance,
123  new FunctionTree(ToValueArray.instance, StreamVariable.X)));
124  Connector.connect(slicer, to_normalized_vector);
125  vector.associateInput(INPUT, slicer, INPUT);
126  vector.associateOutput(OUTPUT, to_normalized_vector, OUTPUT);
127  vector.addProcessors(slicer, to_normalized_vector);
128  }
129 
130  /* Finally, we instantiate a processor mining function, using the
131  * pattern processor defined above, and the K-means clustering
132  * algorithm as the aggregate function. */
133  ProcessorMiningFunction<String, ArrayList<?>> pmf = new ProcessorMiningFunction<String,ArrayList<?>>(vector, new ApplyFunction(new KMeansFunction(2)));
134 
135  /* Let us see the clusters computed by this mining function on the set
136  * of input sequences. There should be two centroids, one roughly
137  * corresponding to a 30-70 distribution of a's vs. b's, and the other
138  * one corresponding to a 70-30 distribution. */
139  Set<?> centroids = (Set<?>) pmf.mine(sequences);
140  System.out.println(centroids);
141  }
142 }
