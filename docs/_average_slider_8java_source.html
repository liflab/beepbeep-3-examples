25 /**
26  * Display a real-time plot of the values of a slider in a Swing frame.
27  * In this example, a <tt>JFrame</tt> containing a simple slider widget
28  * is polled for its value twice per second. This value is then processed
29  * to display a 2D scatterplot with two data series:
30  * <ul>
31  * <li>The value of the slider at each time point</li>
32  * <li>The average value of the slider for the last 4 time points</li>
33  * </ul>
34  * Here is a screenshot of the two JFrames used in this example. At the left
35  * is the slider the user can move, and at the right, the generated plot that
36  * updates once per second. Notice how the use of an average (green line)
37  * smoothes the jittery motion of the raw values (blue line).
38  * <p>
39  * <img src="{@docRoot}/doc-files/widgets/AverageSlider-screenshot.png" width="50%">
40  * <p>
41  * Graphically, the processor chain for this example can be represented
42  * as follows:
43  * <p>
44  * <img src="{@docRoot}/doc-files/widgets/AverageSlider.png" alt="Processor graph"></a>
45  * <p>
46  * As one can see, this example is notable for its mix of various processors:
47  * <ul>
48  * <li>{@link ca.uqac.lif.cep.tmp.CountDecimate CountDecimate}</li>
49  * <li>{@link ca.uqac.lif.cep.functions.Cumulate Cumulate}</li>
50  * <li>{@link ca.uqac.lif.cep.mtnp.DrawPlot DrawPlot}</li>
51  * <li>{@link ca.uqac.lif.cep.tmf.Fork Fork}</li>

71  Connector.connect(pump, fork);
72  ApplyFunction one = new ApplyFunction(new Constant(1));
73  Connector.connect(fork, 0, one, INPUT);
74  Cumulate counter = new Cumulate(new CumulativeFunction<Number>(Numbers.addition));
75  Connector.connect(one, counter);
76  Trim trim1 = new Trim(4);
77  Connector.connect(counter, trim1);
78  Trim trim2 = new Trim(4);
79  Connector.connect(fork, 1, trim2, INPUT);
80  Window win = new Window(new Average(), 4);
81  Connector.connect(fork, 2, win, INPUT);
82 
83  UpdateTable table = new UpdateTableStream("t", "x", "x̅");
84  Connector.connect(counter, OUTPUT, table, 0);
85  Connector.connect(trim2, OUTPUT, table, 1);
86  Connector.connect(win, OUTPUT, table, 2);
87  CountDecimate decimate = new CountDecimate(2);
88  Connector.connect(table, decimate);
89 
90  DrawPlot plot = new DrawPlot(new Scatterplot());
