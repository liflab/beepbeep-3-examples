19  * Group processor that computes the cumulative average of a stream of
20  * numbers. This processor is similar to the {@link basic.Average Average}
21  * examples of the {@link basic} section. Graphically, it corresponds to the
22  * following chain of processors:
23  * <p>
24  * <img
25  * src="./doc-files/widgets/Average.png"
26  * alt="Processor graph">
27  * <p>
28  * Since we encapsulate this chain into a
29  * {@link ca.uqac.lif.cep.GroupProcessor GroupProcessor}, in other processor
30  * graphs that use it, it will be drawn as this:
31  * <p>
32  * <img
33  * src="./doc-files/widgets/Average-box.png"
34  * alt="Processor graph">
35  *
36  * @author Sylvain Hall√©
37  * @see basic.Average Average
38  * @difficulty Easy
39  */
40 public class Average extends GroupProcessor
41 {
42  public Average()
43  {
44  super(1, 1);
45  try
46  {
47  Fork fork = new Fork(2);
48  associateInput(0, fork, 0);
49  FunctionProcessor one = new FunctionProcessor(new Constant(1));
50  Connector.connect(fork, BOTTOM, one, INPUT);
51  CumulativeProcessor count = new CumulativeProcessor(new CumulativeFunction<Number>(Addition.instance));
52  Connector.connect(one, count);
53  CumulativeProcessor sum = new CumulativeProcessor(new CumulativeFunction<Number>(Addition.instance));
54  Connector.connect(fork, TOP, sum, INPUT);
55  FunctionProcessor div = new FunctionProcessor(Division.instance);
56  Connector.connect(sum, OUTPUT, div, TOP);
57  Connector.connect(count, OUTPUT, div, BOTTOM);
58  associateOutput(0, div, 0);
59  addProcessors(fork, one, count, sum, div);
60  }
61  catch (ConnectorException e)
62  {
63  // Do nothing
64  }
65  }
66 }
