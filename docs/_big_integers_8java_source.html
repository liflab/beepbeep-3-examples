25 /**
26  * Utility class to manipulate Java's {@link BigInteger} objects. Since the

72  /**
73  * Computes the square root of a big decimal
74  * @param n The number
75  * @return The square root of this number, rounded to the
76  * nearest integer
77  */
78  public static BigInteger sqrt(BigInteger n)
79  {
80  int firsttime = 0;
81 
82  BigDecimal myNumber = new BigDecimal(n);
83  BigDecimal g = new BigDecimal("1");
84  BigDecimal my2 = new BigDecimal("2");
85  BigDecimal epsilon = new BigDecimal("0.0000000001");
86 
87  BigDecimal nByg = myNumber.divide(g, 9, BigDecimal.ROUND_FLOOR);
88 
89  //Get the value of n/g
90  BigDecimal nBygPlusg = nByg.add(g);
91 
92  //Get the value of "n/g + g
93  BigDecimal nBygPlusgHalf = nBygPlusg.divide(my2, 9, BigDecimal.ROUND_FLOOR);
94 
95  //Get the value of (n/g + g)/2
96  BigDecimal saveg = nBygPlusgHalf;
97  firsttime = 99;
98 
99  do
100  {
101  g = nBygPlusgHalf;
102  nByg = myNumber.divide(g, 9, BigDecimal.ROUND_FLOOR);
103  nBygPlusg = nByg.add(g);
104  nBygPlusgHalf = nBygPlusg.divide(my2, 9, BigDecimal.ROUND_FLOOR);
105  BigDecimal savegdiff = saveg.subtract(nBygPlusgHalf);
106 
107  if (savegdiff.compareTo(epsilon) == -1 )
108  {
109  firsttime = 0;
110  }
111  else
112  {
113  saveg = nBygPlusgHalf;
114  }
115 
116  } while (firsttime > 1);
117  return saveg.toBigInteger();
118  }
119 }
