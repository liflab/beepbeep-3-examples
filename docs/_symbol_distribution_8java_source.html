88 public class SymbolDistribution
89 {
90  @SuppressWarnings("rawtypes")
91  public static void main(String[] args) throws ConnectorException
92  {
93  StreamReader reader = new StreamReader(FileHelper.internalFileToStream(SymbolDistribution.class, "SymbolDistribution.txt"));
94  CsvFeeder feeder = new CsvFeeder();
95  Connector.connect(reader, feeder);
96  GroupProcessor counter = new GroupProcessor(1, 1);
97  {
98  ConstantProcessor one = new ConstantProcessor(new Constant(1));
99  counter.associateInput(INPUT, one, INPUT);
100  CumulativeProcessor sum_one = new CumulativeProcessor(new CumulativeFunction<Number>(Addition.instance));
101  Connector.connect(one, sum_one);
102  counter.associateOutput(OUTPUT, sum_one, OUTPUT);
103  counter.addProcessors(one, sum_one);
104  }
105  SlicerMap slicer = new SlicerMap(new IdentityFunction(1), counter);
106  HashMap<String,Integer> pattern = new HashMap<String,Integer>();
107  pattern.put("a", 6);
108  pattern.put("b", 1);
109  pattern.put("c", 2);
110  TooFarFromTrend<HashMap,Number,Number> alarm = new TooFarFromTrend<HashMap,Number,Number>(pattern, 9, slicer, new FunctionTree(AbsoluteValue.instance,
111  new FunctionTree(MapDistance.instance, new ArgumentPlaceholder(0), new ArgumentPlaceholder(1))), 2, IsLessThan.instance);
112  Connector.connect(feeder, alarm);
113  Pullable p = alarm.getPullableOutput();
114  boolean b = true;
115  for (int i = 0; b && i < 10; i++)
116  {
117  b = (Boolean) p.pull();
118  System.out.println(b);
119  }
120  }
121 }
