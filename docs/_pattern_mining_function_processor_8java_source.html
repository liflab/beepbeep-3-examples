42 /**
43  * Create a mining function from BeepBeep processors.
44  * @see AverageMiningFunction
45  * @author Sylvain Hallé
46  * @difficulty Medium
47  */
48 public class PatternMiningFunctionProcessor
49 {
50  public static void main(String[] args) throws FunctionException
51  {
52  /* First, we must get from somewhere a set of sequences. For the sake
53  * of this example, we just create a few dummy sequences of numbers
54  * from the contents of a file. */
55  Set<Sequence<Number>> sequences = SequenceReader.readNumericalSequences("numbers-1.csv");
56 
57  /* We create a processor that computes the number of times a value
58  * is followed by the same value. */
59  GroupProcessor total_same = new GroupProcessor(1, 1);
60  {
61  Fork fork = new Fork(3);
62  Trim trim = new Trim(1);
63  Connector.connect(fork, 0, trim, INPUT);
64  ApplyFunction equals = new ApplyFunction(Equals.instance);
65  Connector.connect(fork, 1, equals, BOTTOM);
66  Connector.connect(trim, OUTPUT, equals, TOP);
67  ReplaceWith ones = new ReplaceWith(new Constant(1));
68  Connector.connect(fork, 2, ones, INPUT);
69  Filter filter = new Filter();
70  Connector.connect(ones, OUTPUT, filter, TOP);
71  Connector.connect(equals, OUTPUT, filter, BOTTOM);
72  Cumulate sum = new Cumulate(new CumulativeFunction<Number>(Numbers.addition));
73  Connector.connect(filter, sum);
74  total_same.associateInput(INPUT, fork, INPUT);
75  total_same.associateOutput(OUTPUT, sum, OUTPUT);
76  total_same.addProcessors(fork, sum, trim, ones, equals, filter);
77  }
78 
79  /* We then create an instance of our mining function. A mining function
80  * takes as input a set of sequences, and returns for its output some
81  * "pattern" extracted from this set of sequences. */
82  ProcessorMiningFunction<Number,Number> a_f = new ProcessorMiningFunction<Number,Number>(total_same, new Cumulate(new CumulativeFunction<Number>(Numbers.maximum)), 0);
83 
84  /* We then evaluate the function on our set of sequences. */
85  Object[] outputs = new Object[1];
86  a_f.evaluate(new Object[]{sequences}, outputs);
87 
88  /* In this case, the result of this function should be the maximum
89  * number of times the same value has been seen twice in a single
90  * input sequence. */
91  System.out.println(outputs[0]);
92  }
93 
94 }
