26 /**
27  * Show the basic usage of
28  * {@link ca.uqac.lif.cep.functions.Function Function} objects.
29  * {@link ca.uqac.lif.cep.functions.Function#evaluate(Object[], Object[])
30  * @author Sylvain Hallé
31  * @difficulty Easy
32  */
33 public class FunctionUsage
34 {
35  public static void main(String[] args)
36  {
37  /// We create an instance of the negation function. The class
38  // Booleans contains a static reference to an instance of that function
39  Function negation = Booleans.not;
40 
41  // To hold the return value of a function, we must give it an array
42  Object[] out = new Object[1];
43 
44  // Let us evaluate the negation of a Boolean value. This is done
45  // by calling evaluate on the function object.
46  negation.evaluate(new Object[]{true}, out);
47 
48  // We print the return value of the function
49  System.out.println("The return value of the function is: " + out[0]);
50  ///
51 
52  //* Let us now try the same thing with a function that takes
53  // two arguments.
54  Function addition = Numbers.addition;
55 
56  // Since addition takes two arguments, we must pass it an array of
57  // two objects
58  addition.evaluate(new Object[]{2, 3}, out);
59 
60  // We print the return value of the function
61  System.out.println("The return value of the function is: " + out[0]);
62  //*
63 
64  //# Finally, a function does not necessarily have an output arity of 1.
65  // In this example, we call a function that has both an input and an
66  // output arity of 2. From two numbers x and y, it outputs the
67  // quotient and the remainder of the division of x by y.
68  Function int_division = IntegerDivision.instance;
69  Object[] outs = new Object[2];
70  int_division.evaluate(new Object[]{14, 3}, outs);
71  System.out.println("14 divided by 3 equals " +
72  outs[0] + " remainder " + outs[1]);
73  //#
74  }

76  /**
77  * A function that computes integer division. From two numbers x and y,
78  * it outputs the quotient and the remainder of the division of x by y.
79  * This function is used in the code example above to show that functions
80  * can have an output arity different from 1.
81  */
82  public static class IntegerDivision extends Function
83  {
84  /**
85  * A reference to a single publicly accessible instance of the
86  * function
87  */
88  public static final transient IntegerDivision instance = new IntegerDivision();
89 
90  /**
91  * The constructor is declared private on purpose
92  */
93  private IntegerDivision()
94  {
95  super();
96  }
97 
98  @Override
99  public Function duplicate(boolean with_state)
100  {
101  return instance;
102  }
103 
104  @Override
105  public void evaluate(Object[] inputs, Object[] outputs)
106  {
107  int x = (Integer) inputs[0];
108  int y = (Integer) inputs[1];
109  outputs[0] = x / y;
110  outputs[1] = x % y;
111  }
112 
113  @Override

117  }

123  }
124

126  public void reset()
