42 /**
43  * The code for Machine A in the twin prime example.
44  * @author Sylvain Hallé
45  */
46 public class TwinPrimesA
47 {
48  public static void main(String[] args) throws ProcessorException
49  {
50  ///
51  /* The URL where prime numbers will be pushed downstream. Change
52  * this string to correspond to Machine B's address and port. */
53  String push_url = "http://localhost:12312/bigprime";
54 
55  /* The first processor is a source that will push the BigInteger
56  * "2" repeatedly. */
57  QueueSource source = new QueueSource().addEvent(new BigInteger("2"));
58 
59  /* We the connect a pump to that source. */
60  Pump pump = new Pump(500);
61  Connector.connect(source, pump);
62 
63  /* The second processor is a simple counter. We will feed it with the
64  * BigInteger "2" repeatedly, and it will return the cumulatve sum
65  * of those "2" as its output. Since the start value of BigIntegerAdd
66  * is one, the resulting sequence is made of all odd numbers. */
67  Cumulate counter = new Cumulate(
68  new CumulativeFunction<BigInteger>(BigIntegerAdd.instance));
69  Connector.connect(pump, counter);
70 
71  /* The events output from the counter are duplicated along two paths. */
72  Fork fork1 = new Fork(2);
73  Connector.connect(counter, fork1);
74 
75  /* Along the first path, the numbers are checked for primality. */
76  ApplyFunction prime_check = new ApplyFunction(IsPrime.instance);
77  Connector.connect(fork1, LEFT, prime_check, INPUT);
78 
79  /* Along the second path, we feed a filter and use the primality
80  * verdict as the filtering condition. What comes out of the filter
81  * are only prime numbers. */
82  Filter filter = new Filter();
83  Connector.connect(fork1, RIGHT, filter, LEFT);
84  Connector.connect(prime_check, OUTPUT, filter, RIGHT);
85 
86  /* We fork the output of the filter, just so that we can
87  * print what comes out of it. */
88  Fork fork2 = new Fork(2);
89  Connector.connect(filter, fork2);
90  Print print = new Print();
91  Connector.connect(fork2, LEFT, print, INPUT);
92 
93  /* We convert BigIntegers to Strings, and push them across the network
94  * to Machine B using the HttpUpstreamGateway. */
95  ApplyFunction int_to_string =
96  new ApplyFunction(BigIntegerToString.instance);
97  HttpUpstreamGateway up_gateway = new HttpUpstreamGateway(push_url);
98  Connector.connect(fork2, RIGHT, int_to_string, INPUT);
99  Connector.connect(int_to_string, up_gateway);
100  ///
101 
102  /* All set! We are ready to start the gateway, and repeatedly push
103  * primes to Machine B. */
104  System.out.println("This is Machine A. Press Enter to start pushing numbers to Machine B.");
105  UtilityMethods.readLine();
106  System.out.println("Let's go! Pushing prime numbers to " + push_url);
107  System.out.println("Press Ctrl+C to stop.");
108  up_gateway.start();
109  pump.start();
110  }
111 }
