27 /**
28  * Pipe processors together using the {@link ca.uqac.lif.cep.Connector Connector}
29  * object.
30  * The chain of processors in this example can be represented
31  * graphically as:
32  * <p>
33  * <img src="./doc-files/basic/PipingBinary.png" alt="Processor graph">
34  * @see PipingUnary
35  * @author Sylvain Hallé
36  * @difficulty Easy
37  */
38 public class PipingBinary
39 {
40  public static void main (String[] args) throws ConnectorException
41  {
42  QueueSource source1 = new QueueSource();
43  source1.setEvents(new Integer[]{2, 7, 1, 8, 3});
44  QueueSource source2 = new QueueSource();
45  source2.setEvents(new Integer[]{3, 1, 4, 1, 6});
46  FunctionProcessor add = new FunctionProcessor(Addition.instance);
47  Connector.connect(source1, 0, add, 0);
48  Connector.connect(source2, 0, add, 1);
49  Pullable p = add.getPullableOutput();
50  for (int i = 0; i < 5; i++)
51  {
52  float x = (Float) p.pull();
53  System.out.println("The event is: " + x);
54  }
55  }
56 }
