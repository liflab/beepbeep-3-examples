33 /**
34  * Calculate the Euclidean distance of each two successive points in an
35  * input trace of (x,y) coordinates. For two points (<i>x</i><sub>1</sub>,<i>y</i><sub>1</sub>)
36  * and (<i>x</i><sub>2</sub>,<i>y</i><sub>2</sub>), the Euclidean distance is
37  * defined as the square root of (<i>x</i><sub>1</sub>-<i>x</i><sub>2</sub>)<sup>2</sup> +
38  * (<i>y</i><sub>1</sub>-<i>y</i><sub>2</sub>)<sup>2</sup>. The chain of
39  * processors in this example can be represented graphically as:
40  * <p>
41  * <img src="./doc-files/basic/PointDistance.png" alt="Processor graph">
42  * <p>
43  * In this picture, light green pipes correspond to streams of {@link Point}
44  * objects. The processor with a ruler and a "d" is the {@link Distance}
45  * processor defined below.
46  * <p>
47  * On an input stream made of points (2,7), (1,8), (2,8), (1,8), &hellip;, the
48  * expected output of this program should look like:
49  * <pre>
50  * 1.4142135
51  * 1.0
52  * 1.0
53  * 1.0
54  * 3.6055512
55  * &hellip;
56  * </pre>
57  *
58  * @author Sylvain Hallé
59  * @difficulty Easy
60  */
61 public class PointDistance
62 {
63  /* We create a new type of event: Point. A point is just
64  * an "x" and a "y" value. This shows that in BeepBeep, anything can be an
65  * event, including user-defined objects. */
66  public static class Point
67  {
68  public float x;
69  public float y;
70 
71  public Point(float x, float y)
72  {
73  this.x = x;
74  this.y = y;
75  }
76  }
77 
78  /**
79  * We create a new function: Distance. This function takes two points
80  * as its input, and returns their Euclidean distance as its output.
81  * It is therefore a 2:1 function.
82  *
83  * @see functions.AddNumbers#AddNumbers()
84  */
85  public static class Distance extends BinaryFunction<Point,Point,Float>
86  {
87  public Distance()
88  {
89  super(Point.class, Point.class, Float.class);
90  }
91 
92  @Override
93  public Float getValue(Point p1, Point p2)
94  {
95  return (float) Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
96  }
97  }
98 
99  /*
100  * A small main to illustrate the concept.
101  */
102  public static void main(String[] args) throws ConnectorException
103  {
104  /* Create a new source made of arbitrary Point objects. */
105  QueueSource point_source = new QueueSource(1);
106  point_source.setEvents(getListOfPoints());
107 
108  /* Duplicate the output of source along two paths. */
109  Fork fork = new Fork(2);
110  connect(point_source, OUTPUT, fork, INPUT);
111 
112  /* The first path is plugged directly as the first argument of a
113  * function processor that computes the Euclidean distance between
114  * two points. */
115  FunctionProcessor distance_proc = new FunctionProcessor(new Distance());
116  connect(fork, LEFT, distance_proc, LEFT);
117 
118  /* The second path is trimmed of its first event... */
119  Trim trim = new Trim(1);
120  connect(fork, RIGHT, trim, INPUT);
121 
122  /* And is then plugged as the second input of the distance processor.
123  * This has for effect that distance_proc will compute the distance
124  * between each point and the next. */
125  connect(trim, OUTPUT, distance_proc, RIGHT);
126 
127  /* Pull and print the first 10 events from the output. */
128  Pullable p = distance_proc.getPullableOutput(OUTPUT);
129  for (int i = 0; i < 10; i++)
130  {
131  float d = (Float) p.pull();
132  System.out.println(d);
133  }
134  }
