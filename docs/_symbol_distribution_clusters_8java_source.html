50 /**
51  * Trend distance based on the statistical distribution of symbols in a
52  * stream. In this example, a feature vector is computed from an input
53  * trace by calculating the fraction of a's and b's that occur in a sliding
54  * window of width 9
55  * (see {@link mining.extraction.KmeansSymbolDistribution KmeansSymbolDistribution}
56  * for an explanation of how this is computed).
57  * The reference pattern is a set of two-dimensional points, corresponding
58  * to the centroids of two clusters. The distance function computes the
59  * Euclidean distance between the computed feature vector and the
60  * <em>closest</em> centroid of the reference set. If this distance is greater
61  * than <i>d</i>=0.15, an alarm is raised.
62  * <p>
63  * For example, suppose that the two centroids have coordinates (0.7, 0.3) and
64  * (0.3, 0.7); they are represented by two crosses in the 2D plot below.
65  * <p>
66  * <img src="./doc-files/mining/extraction/ClusteringAB.png" alt="Plot">
67  * <p>
68  * Consider the following window of 9 events:
69  * <pre>
70  * a, b, a, b, a, b, a, b, a
71  * </pre>
72  * The feature vector extracted from this window is (0.56, 0.44) (red dot
73  * in the plot above). The centroid closest to this point is (0.7, 0.3),
74  * but its distance is 0.2, which is greater than 0.15. In that case, the
75  * feature vector is considered "too far" from existing clusters, and
76  * an alarm is raised.
77  * <p>
78  * The parameters of the <tt>TrendDistance</tt> processor in this example
79  * are as follows:
80  * <table>
81  * <tr><th>Parameter</th><th>Value</th></tr>
82  * <tr>
83  * <td><img src="./doc-files/mining/trenddistance/WidthParameter.png" alt="Window Width" title="The width of the window"></td>
84  * <td>9</td>
85  * </tr>
86  * <tr>
87  * <td><img src="./doc-files/mining/trenddistance/BetaProcessor.png" alt="Beta processor" title="The processor that computes the pattern over the current input stream"></td>
88  * <td><img src="./doc-files/mining/extraction/SymbolDistributionDoublePoint.png" alt="Processor chain"></td>
89  * </tr>
90  * <tr>
91  * <td><img src="./doc-files/mining/trenddistance/PatternParameter.png" alt="Reference Pattern" title="The reference pattern"></td>
92  * <td>{(0.7, 0.3), (0.3, 0.7)}</td>
93  * </tr>
94  * <tr>
95  * <td><img src="./doc-files/mining/trenddistance/DistanceFunction.png" alt="Distance Function" title="The function that computes the distance with respect to the reference pattern"></td>
96  * <td><img src="./doc-files/mining/trenddistance/DistanceToClosest.png" alt="Distance Function">
97  * ({@link ca.uqac.lif.cep.peg.ml.DistanceToClosest DistanceToClosest}
98  * using {@link org.apache.commons.math3.ml.distance.EuclideanDistance EuclideanDistance} metric)</td>
99  * </tr>
100  * <tr>
101  * <td><img src="./doc-files/mining/trenddistance/ComparisonFunction.png" alt="Comparison Function" title="The function that compares that distance with a given threshold"></td>
102  * <td><img src="./doc-files/mining/LessThanOrEqual.png" alt="&leq;"></td>
103  * </tr>
104  * <tr>
105  * <td><img src="./doc-files/mining/trenddistance/DistanceThreshold.png" alt="Distance Threshold" title="The distance threshold"></td>
106  * <td>¼</td>
107  * </tr>
108 
109  * </table>
110  *
111  * @author Sylvain Hallé
112  *
113  */
114 public class SymbolDistributionClusters
115 {
116  public static void main(String[] args)
117  {
118  ReadStringStream reader = new ReadStringStream(SymbolDistributionClusters.class.getResourceAsStream("SymbolDistribution-AB.txt"));
119  FindPattern feeder = new FindPattern("(.*?),");
120  Connector.connect(reader, feeder);
121  /* We then create a processor that computes the feature vector
122  * from an input trace. */
123  GroupProcessor vector = new GroupProcessor(1, 1);
124  {
125  GroupProcessor counter = new GroupProcessor(1, 1);
126  {
127  ReplaceWith one = new ReplaceWith(new Constant(1));
128  counter.associateInput(INPUT, one, INPUT);
129  Cumulate sum_one = new Cumulate(new CumulativeFunction<Number>(Numbers.addition));
130  Connector.connect(one, sum_one);
131  counter.associateOutput(OUTPUT, sum_one, OUTPUT);
132  counter.addProcessors(one, sum_one);
133  }
134  Slice slicer = new Slice(new IdentityFunction(1), counter);
135  ApplyFunction to_normalized_vector = new ApplyFunction(
136  new FunctionTree(DoublePointCast.instance,
137  new FunctionTree(Normalize.instance,
138  new FunctionTree(ToValueArray.instance, StreamVariable.X))));
139  Connector.connect(slicer, to_normalized_vector);
140  vector.associateInput(INPUT, slicer, INPUT);
141  vector.associateOutput(OUTPUT, to_normalized_vector, OUTPUT);
142  vector.addProcessors(slicer, to_normalized_vector);
143  }
144  Connector.connect(feeder, vector);
145  Set<DoublePoint> pattern = new HashSet<DoublePoint>();
146  pattern.add(new DoublePoint(new double[]{0.7, 0.3}));
147  pattern.add(new DoublePoint(new double[]{0.3, 0.7}));
148  TrendDistance<Set<DoublePoint>,Set<DoublePoint>,Number> alarm = new TrendDistance<Set<DoublePoint>,Set<DoublePoint>,Number>(pattern, 9, vector, new FunctionTree(Numbers.absoluteValue,
149  new FunctionTree(new DistanceToClosest(new EuclideanDistance()), StreamVariable.X, StreamVariable.Y)), 0.25, Numbers.isLessThan);
150  Connector.connect(feeder, alarm);
151  Pullable p = alarm.getPullableOutput();
152  boolean b = true;
153  for (int i = 0; b && i < 10; i++)
154  {
155  b = (Boolean) p.pull();
156  System.out.println(b);
157  }
158  }
159 }
