31 /**
32  * Computes the sum of a sliding window of successive Fibonacci numbers.
33  * The goal of this program is to compare its performance with that of a
34  * similar one called {@link WindowSequential}. Both perform the same
35  * computation. However, in the present program, the computation of the window
36  * processor is done *sequentially*. That is, upon receiving an event,
37  * the window processor pushes it to each window one after the other in the
38  * current thread.
39  * The output of the processor chain is not printed anywhere; rather, the
40  * program simply counts the *time* taken to process the inputs.
41  *
42  * On a machine with multiple cores, this program should run slower than
43  * {@link WindowParallel} for the same values of
44  * {@link WindowParallel#WIDTH WIDTH} and
45  * {@link WindowParallel#NUM_WINDOWS NUM_WINDOWS}. Note that the performance
46  * gain may become apparent only for large values of these two parameters.
47  * For a small number of small windows, the overhead of using multiple threads
48  * offsets the potential performance gains offered by concurrency.
49  *
50  * @see WindowParallel
51  */
52 public class WindowSequential
53 {
54  public static void main(String[] args) throws InterruptedException
55  {
56  // Create a group processor. From an input event n, it computes the
57  // n-th Fibonacci number and sends it to a cumulative sum processor.
58  GroupProcessor fib_sum = new GroupProcessor(1, 1);
59  {
60  FibonacciProcessor fib = new FibonacciProcessor();
61  Cumulate sum = new Cumulate(new CumulativeFunction<BigInteger>(BigIntegers.addition));
62  Connector.connect(fib, sum);
63  fib_sum.addProcessors(fib, sum);
64  fib_sum.associateInput(0, fib, 0);
65  fib_sum.associateOutput(0, sum, 0);
66  }
67  // Give this processor as the argument to a Window processor
68  ParallelWindow win = new ParallelWindow(fib_sum, WindowParallel.WIDTH);
69  // Connect this window to a black hole. In this example, we are not
70  // interested in the actual values produced by the processor chain,
71  // only in the *time* taken to compute those values.
72  BlackHole print = new BlackHole();
73  Connector.connect(win, print);
74  // Push NUM_WINDOWS events
75  Pushable p = win.getPushableInput();
76  Stopwatch sw = new Stopwatch();
77  sw.start();
78  for (int i = 2; i < WindowParallel.NUM_WINDOWS; i++)
79  {
80  p.push(i);
81  }
82  // Wait for these events to be processed and show the running time
83  sw.stop();
84  System.out.println("All windows have been computed in " + sw.elapsed() + " ms");
85  }
86 }
