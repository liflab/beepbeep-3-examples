31  // Fork the stream in two and connect it to the source
32  GroupProcessor group = new GroupProcessor(1, 1);

41  group.associateInput(0, fork, 0);
42  group.associateOutput(0, add, 0);
43  }
44 
45  // Connect the source to the group
