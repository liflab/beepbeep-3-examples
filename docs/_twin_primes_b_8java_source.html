40 /**
41  * The code for Machine B in the twin prime example.
42  * @author Sylvain Hallé
43  */
44 public class TwinPrimesB
45 {
46  public static void main(String[] args) throws ProcessorException
47  {
48  /* First, we create an HttpDownstreamGateway to receive strings
49  * from Machine A. */
50  ///
51  HttpDownstreamGateway dn_gateway = new HttpDownstreamGateway(12312, "/bigprime", Method.POST);
52 
53  /* The next step is to convert the string received from the gateway
54  * back into a BigInteger. We then increment this number by 2. We do
55  * this with a special function. */
56  ApplyFunction big_int_plus_2 = new ApplyFunction(new FunctionTree(
57  new BigIntegerFunctions.IncrementBigInteger(new BigInteger("2")),
58  new FunctionTree(StringToBigInteger.instance, StreamVariable.X)));
59  Connector.connect(dn_gateway, big_int_plus_2);
60 
61  /* Fork the output */
62  Fork fork = new Fork(2);
63  Connector.connect(big_int_plus_2, fork);
64 
65  /* Primality check for n+2... */
66  ApplyFunction is_prime = new ApplyFunction(IsPrime.instance);
67  Connector.connect(fork, LEFT, is_prime, INPUT);
68 
69  /* Join both forks into a filter. What comes out is a stream of numbers
70  * of the form n+2, such that both n and n+2 are prime. */
71  Filter filter = new Filter();
72  Connector.connect(fork, RIGHT, filter, LEFT);
73  Connector.connect(is_prime, OUTPUT, filter, RIGHT);
74 
75  /* Print n+2 */
76  Print print = new Print();
77  Connector.connect(filter, print);
78 
79  /* All set! Start the gateway so it can listen to requests from
80  * Machine A. */
81  dn_gateway.start();
82  ///
83 
84  /* Loop indefinitely */
85  System.out.println("Machine B listening for requests. Every number displayed below");
86  System.out.println("is the second of a twin prime pair. Press Ctrl+C to end.");
87  while (true)
88  {
89  UtilityMethods.pause(10000);
90  }
91  }
92 }
