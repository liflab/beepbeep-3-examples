30 /**
31  * Pipe processors together using the {@link ca.uqac.lif.cep.Connector Connector}
32  * object. The chain of processors in this example can be represented
33  * graphically as:
34  * <p>
35  * <img src="./doc-files/basic/PipingUnary.png" alt="Processor graph">
36  * @see PipingBinary#main(String[])
37  * @author Sylvain Hallé
38  */
39 public class PipingUnary
40 {
41  public static void main (String[] args)
42  {
43  /* Create a source of arbitrary numbers. */
44  QueueSource source = new QueueSource();
45  source.setEvents(new Integer[]{1, 2, 3, 4, 5, 6});
46 
47  /* Create an instance of the Doubler processor (which is defined just
48  * below in this file. */
49  Doubler doubler = new Doubler();
50 
51  /* We now want to connect the output of source to the input of doubler.
52  * This connection is done by using static method connect() of the
53  * Connector object. Here, source has only one output stream, and
54  * doubler has only one input stream. Therefore, we can simply call
55  * connect() by giving the "upstream" processor first, and the
56  * "downstream" processor second. The Connector will know that it needs
57  * to connect the (only) output of source to the (only) input of
58  * doubler. */
59  try
60  {
61  Connector.connect(source, doubler);
62  }
63  catch (ConnectorException e)
64  {
65  /* We must surround this call by a try/catch block. The connect()
66  * method may throw a ConnectorException, indicating that the piping
67  * could not be done. This can happen for various reasons:
68  *
69  * - The output type of the upstream processor is incompatible with
70  * the input type of the downstream processor.
71  * - The call refers to an input or output stream that does not
72  * exist. For example, it would be an error to attempt to connect
73  * the *second* output of source to something, as there is no second
74  * output stream on this processor.
75  *
76  * Here however, everything is legit, so no exception should be thrown.
77  */
78  e.printStackTrace();
79  }
80 
81  /* Let us now print what we receive by pulling on the output of
82  * doubler. */
83  Pullable p = doubler.getPullableOutput();
84  for (int i = 0; i < 8; i++)
85  {
86  int x = (Integer) p.pull();
87  System.out.println("The event is: " + x);
88  // Sleep a little so you have time to read
89  UtilityMethods.pause(1000);
90  }
91  }
92 
93  /**
94  * A processor that doubles every number it is given. What this
95  * processor does is not really important for our example.
96  */
97  public static class Doubler extends SingleProcessor
98  {
99  public Doubler()
100  {
101  super(1, 1);
102  }
103 
104  @Override
105  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
106  {
107  outputs.add(wrapObject(2 * ((Number) inputs[0]).intValue()));
108  return true;
109  }
110 
111  @Override
112  public Processor clone()
113  {
114  return this;
115  }
116  }
117 }
