29 /**
30  * Pipe processors together using the {@link ca.uqac.lif.cep.Connector Connector}
31  * object. The chain of processors in this example can be represented
32  * graphically as:
33  * <p>
34  * <img src="{@docRoot}/doc-files/basic/PipingUnary.png" alt="Processor graph">
35  * @see PipingBinary
36  * @author Sylvain Hallé
37  * @difficulty Easy
38  */
39 public class PipingUnary
40 {
41  public static void main (String[] args)
42  {
43  /// Create a source of arbitrary numbers
44  QueueSource source = new QueueSource();
45  source.setEvents(1, 2, 3, 4, 5, 6);
46 
47  /* Create an instance of the Doubler processor (which is defined just
48  * below in this file. */
49  Doubler doubler = new Doubler();
50 
51  /* We now want to connect the output of source to the input of doubler.
52  * This connection is done by using static method connect() of the
53  * Connector object. Here, source has only one output stream, and
54  * doubler has only one input stream. Therefore, we can simply call
55  * connect() by giving the "upstream" processor first, and the
56  * "downstream" processor second. The Connector will know that it needs
57  * to connect the (only) output of source to the (only) input of
58  * doubler. */
59  Connector.connect(source, doubler);
60 
61  /* Let us now print what we receive by pulling on the output of
62  * doubler. */
63  Pullable p = doubler.getPullableOutput();
64  for (int i = 0; i < 8; i++)
65  {
66  int x = (Integer) p.pull();
67  System.out.println("The event is: " + x);
68  // Sleep a little so you have time to read
69  UtilityMethods.pause(1000);
70  }
71  ///
72  }
73 
74  /**
75  * A processor that doubles every number it is given. What this
76  * processor does is not really important for our example.
77  */
78  public static class Doubler extends SingleProcessor
79  {
80  public Doubler()
81  {
82  super(1, 1);
83  }
84 
85  @Override
86  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
87  {
88  outputs.add(wrapObject(2 * ((Number) inputs[0]).intValue()));
89  return true;
90  }
91 
92  @Override
93  public Processor duplicate(boolean with_state)
94  {
95  return this;
96  }
97  }
98 }
