32 /**
33  * Detect planetary encounters of
34  * <a href="https://en.wikipedia.org/wiki/Voyager_2">Voyager 2</a> by analyzing
35  * its distance from the Earth.
36  * <p>
37  * In this example, we read a processed data feed that provides averaged
38  * hourly readings of various instruments in the spacecraft. We are
39  * interested in the date and the relative distance to Earth, expressed in
40  * astronomical units (AU). From these two data fields, we compute the
41  * average weekly speed (in AU/week), and send the output stream to signal
42  * processors that detect peaks (i.e. abrupt variations in the values of
43  * a numerical signal). This is shown in a plot.
44  * <p>
45  * One can see that the last three peaks correspond precisely to the dates of
46  * Voyager's flybys of Jupiter, Saturn, and Neptune:
47  * <p>
48  * <table>
49  * <tr><td>Planet</td><td>Date</td><td>Days after 77/1/1</td></tr>
50  * <tr><td>Jupiter</td><td>July 9, 1979</td><td>918</td></tr>
51  * <tr><td>Saturn</td><td>August 25, 1981</td><td>1696</td></tr>
52  * <tr><td>Neptune</td><td>August 25, 1989</td><td>4618</td></tr>
53  * </table>
54  * <p>
55  * The flyby of Uranus (Jaunary 24, 1986, or Day 3310) does not produce
56  * a speed variation large enough to be detected through this method.
57  *
58  * <h4>About the data files</h4>
59  * <p>This example requires you to first download some data files from the
60  * Voyager 2 space probe; these files are publicly available on a NASA FTP
61  * site:
62  * <p>
63  * <a href="ftp://spdf.gsfc.nasa.gov/pub/data/voyager/voyager2/merged/">ftp://spdf.gsfc.nasa.gov/pub/data/voyager/voyager2/merged/</a>
64  * <p>
65  * @author Sylvain Hall√©
66  */
67 public class PlotSpeed
68 {
69  public static void main(String[] args)
70  {
71  int start_year = 1977, end_year = 1992;
72 
73  /* The Voyager data is split into multiple CSV files, one per civil
74  * year. Let us create one ReadLines processor for each file. */
75  ReadLines[] readers = new ReadLines[end_year - start_year + 1];
76  for (int y = start_year; y <= end_year; y++)
77  {
78  readers[y - start_year] = new ReadLines(PlotSpeed.class.getResourceAsStream("data/vy2_" + y + ".asc"));
79  }
80 
81  /* Pass this array of readers to the Splice processors, so that their
82  * contents be used as an uninterrupted stream of events. */
83  Splice spl = new Splice(readers);
84 
85  /* The input files have hourly readings; keep only one reading per week. */
86  CountDecimate cd = new CountDecimate(24 * 7);
87  Connector.connect(spl, cd);
88 
89  /* The 1977 file has no data before week 31 or so (the launch date);
90  * let's ignore these first lines. */
91  Trim ignore_beginning = new Trim(31);
92  Connector.connect(cd, ignore_beginning);
93 
94  /* Split the surviving lines into arrays with the TAB character. */
95  ApplyFunction to_array = new ApplyFunction(new Strings.SplitString("\\s+"));
96  Connector.connect(ignore_beginning, to_array);
97 
98  /* Fork this stream of arrays in three. */
99  Fork fork = new Fork(3);
100  Connector.connect(to_array, fork);
101 
102  /* From first stream, format date */
103  ApplyFunction format_date = new ApplyFunction(new FunctionTree(FormatDate.instance, new FunctionTree(new NthElement(0), StreamVariable.X), new FunctionTree(new NthElement(1), StreamVariable.X)));
104  Connector.connect(fork, 0, format_date, INPUT);
105 
106  /* From second stream, extract distance */
107  ApplyFunction get_au1 = new ApplyFunction(new FunctionTree(Numbers.numberCast, new FunctionTree(new NthElement(3), StreamVariable.X)));
108  Connector.connect(fork, 1, get_au1, INPUT);
109 
110  /* Delay third stream by one reading, and extract distance */
111  Trim cd_delay = new Trim(1);
112  Connector.connect(fork, 2, cd_delay, INPUT);
113  ApplyFunction get_au2 = new ApplyFunction(new FunctionTree(Numbers.numberCast, new FunctionTree(new NthElement(3), StreamVariable.X)));
114  Connector.connect(cd_delay, get_au2);
115 
116  /* Subtract third and second streams, giving the weekly distance
117  * travelled. */
118  ApplyFunction distance = new ApplyFunction(new FunctionTree(Numbers.maximum, Constant.ZERO, new FunctionTree(Numbers.subtraction, StreamVariable.X, StreamVariable.Y)));
119  Connector.connect(get_au2, OUTPUT, distance, TOP);
120  Connector.connect(get_au1, OUTPUT, distance, BOTTOM);
121 
122  /* Since the weekly distance is very close to the measurement's
123  * precision, smoothen those values by averaging each two successive
124  * points. */
125  Smoothen smooth = new Smoothen(12);
126  Connector.connect(distance, smooth);
127 
128  Fork f2 = new Fork(2);
129  Connector.connect(smooth, f2);
130  PeakFinderLocalMaximum peak = new PeakFinderLocalMaximum(5);
131  //Passthrough peak = new Passthrough();
132  Connector.connect(f2, BOTTOM, peak, INPUT);
133  Threshold th = new Threshold(0.0125f);
134  Connector.connect(peak, th);
135  Limiter li = new Limiter(5);
136  Connector.connect(th, li);
137 
138  UpdateTableStream table = new UpdateTableStream("Date", "Speed (AU/week)", "Peak");
139 
140  Connector.connect(format_date, OUTPUT, table, 0);
141  Connector.connect(f2, OUTPUT, table, 1);
142  Connector.connect(li, OUTPUT, table, 2);
143 
144  Pump pump = new Pump();
145  Connector.connect(table, pump);
146 
147  KeepLast last = new KeepLast(1);
148  Connector.connect(pump, last);
149  Scatterplot plot = new Scatterplot();
150  plot.setTitle("Planetary encounters of Voyager 2");
151  plot.setCaption(Axis.X, "Days after 1/1/77");
152  plot.setCaption(Axis.Y, "AU");
153  DrawPlot draw = new DrawPlot(plot);
154  Connector.connect(last, draw);
155 
156  BitmapJFrame window = new BitmapJFrame();
157  Connector.connect(draw, window);
158  window.start();
159  pump.start();
160  }
