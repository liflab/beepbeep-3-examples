35 /**
36  * Compute the sum of every pair of successive events. Here we are
37  * interested in computing the sum of events at position 0-1, 2-3, 4-5, etc.
38  * To do so, we compute first the sum of every two successive events,
39  * and then keep every other event of the resulting trace.
40  * Graphically, this chain of processors can be represented as:
41  * <p>
42  * <img src="./doc-files/basic/Decimation.png" alt="Processor graph">
43  * <p>
44  * On the input stream 6, 5, 3, 8, 9, 2, 1, 7, 4, 5, &hellip;,
45  * the expected output of this program is:
46  * <pre>
47  * Event #0 is: 11
48  * Event #1 is: 11
49  * Event #2 is: 11
50  * Event #3 is: 8
51  * Event #4 is: 9
52  * Event #5 is: 6
53  * Event #6 is: 13
54  * Event #7 is: 20
55  * Event #8 is: 7
56  * </pre>
57  *
58  * @author Sylvain Hall√©
59  */
60 public class Decimation
61 {
62  public static void main(String[] args) throws ConnectorException
63  {
64  /* Create a stream of dummy values. */
65  QueueSource source_values = new QueueSource();
66  source_values.setEvents(new Integer[]{6, 5, 3, 8, 9, 2, 1, 7, 4, 5,
67  2, 4, 7, 6, 12, 8, 1});
68 
69  /* Duplicate this stream into two paths. */
70  Fork fork = new Fork(2);
71  connect(source_values, fork);
72 
73  /* The first path is plugged directly as the first argument of
74  * an adding processor. */
75  FunctionProcessor sum = new FunctionProcessor(Addition.instance);
76  connect(fork, LEFT, sum, LEFT);
77 
78  /* Along the second path, we start by removing the first event of
79  * the stream. This is done using the Trim processor. */
80  Trim trim = new Trim(1);
81  connect(fork, RIGHT, trim, INPUT);
82 
83  /* We then plug the output of this Trim processor as the second
84  * argument of the adding processor. This has for effect that the
85  * sum processor will add events of the input stream at positions
86  * i and i+1 (for i = 0, 1, 2, etc.). From the stream of numbers
87  * defined above, the output of this processor will be:
88  * 6+5, 5+3, 3+8, 8+9, etc. */
89  connect(trim, OUTPUT, sum, RIGHT);
90 
91  /* Since we want to keep only the sums of events at positions 0-1,
92  * 2-3, 4-5, etc., we have to remove from this output every other
93  * event. This is done with the CountDecimate processor, which is
94  * instructed to keep every <i>n</i>-th event it receives; here, n=2. */
95  CountDecimate decimate = new CountDecimate(2);
96  connect(sum, OUTPUT, decimate, INPUT);
97 
98  /* Get the Pullable of this last processor, and show the first few
99  * values it outputs. */
100  Pullable p = decimate.getPullableOutput();
101  for (int i = 0; i < 9; i++)
102  {
103  int v = ((Number) p.pull()).intValue();
104  System.out.printf("Event #%d is: %d\n", i, v);
105  // Pause between each so you have time to read!
106  UtilityMethods.pause(1000);
107  }
108  }
109 }
