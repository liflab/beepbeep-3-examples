42 public class TwinPrimesB
43 {
44  public static void main(String[] args) throws ConnectorException, ProcessorException, InterruptedException
45  {
46  /* First, we create an HttpDownstreamGateway to receive strings
47  * from Machine A. */
48  HttpDownstreamGateway dn_gateway = new HttpDownstreamGateway(12312, "/bigprime", Method.POST);
49 
50  /* The next step is to convert the string received from the gateway
51  * back into a BigInteger. */
52  FunctionProcessor string_to_bigint = new FunctionProcessor(StringToBigInteger.instance);
53  Connector.connect(dn_gateway, string_to_bigint);
54 
55  /* We then increment this number by 2. We do this with a special
56  * function. */
57  FunctionProcessor big_int_plus_2 = new FunctionProcessor(new BigIntegerFunctions.IncrementBigInteger(new BigInteger("2")));
58  Connector.connect(string_to_bigint, big_int_plus_2);
59 
60  /* Fork the output */
61  Fork fork = new Fork(2);
62  Connector.connect(big_int_plus_2, fork);
63 
64  /* Primality check for n+2... */
65  FunctionProcessor is_prime = new FunctionProcessor(IsPrime.instance);
66  Connector.connect(fork, LEFT, is_prime, INPUT);
67 
68  /* Join both forks into a filter. What comes out is a stream of numbers
69  * of the form n+2, such that both n and n+2 are prime. */
70  Filter filter = new Filter();
71  Connector.connect(fork, RIGHT, filter, LEFT);
72  Connector.connect(is_prime, OUTPUT, filter, RIGHT);
73 
74  /* Print n+2 */
75  Print print = new Print();
76  Connector.connect(filter, print);
77 
78  /* All set! Start the gateway so it can listen to requests from
79  * Machine A. */
80  dn_gateway.start();
81 
82  /* Loop indefinitely */
83  System.out.println("Machine B listening for requests. Every number displayed below");
84  System.out.println("is the second of a twin prime pair. Press Ctrl+C to end.");
85  while (true)
86  {
87  Thread.sleep(10000);
88  }
89  }
90 }
