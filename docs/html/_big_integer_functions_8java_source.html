95  /**
96  * Checks if a BigInteger is prime. This function uses a very slow method.
97  */
98  public static class IsPrime extends UnaryFunction<BigInteger,Boolean>
99  {
100  public static final transient IsPrime instance = new IsPrime();
101  private static final transient BigInteger TWO = new BigInteger("2");
102  private static final transient BigInteger THREE = new BigInteger("3");
103 
104  IsPrime()
105  {
106  super(BigInteger.class, Boolean.class);
107  }
108 
109  @Override
110  public Boolean getValue(BigInteger n) throws FunctionException
111  {
112  if (n.compareTo(BigInteger.ONE) == 0 || n.compareTo(TWO) == 0)
113  {
114  return true;
115  }
116  BigInteger half=n.divide(TWO);
117  for (BigInteger i = THREE; i.compareTo(half) <= 0; i = i.add(TWO))
118  {
119 
120  if (n.mod(i).equals(BigInteger.ZERO))
121  {
122  return false;
123  }
124  }
125  return true;
126  }
127  }
