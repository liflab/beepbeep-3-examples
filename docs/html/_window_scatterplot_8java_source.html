31 /**
32  * Update a windowed 2D scatterplot in realtime from two streams of numbers.
33  *
34  * @author Sylvain Hallé
35  */
36 public class WindowScatterplot
37 {
38  public static void main(String[] args) throws ConnectorException, InterruptedException
39  {
40  QueueSource one = new QueueSource();
41  one.addEvent(1);
42  RandomTwoD random = new RandomTwoD();
43  Connector.connect(one, random);
44  NaryToArray array_convert = new NaryToArray(2);
45  Connector.connect(random, array_convert);
46 
47  /* Apply a sliding window */
48  Window window = new Window(new UpdateTableArray("x", "y"), 100);
49  Connector.connect(array_convert, window);
50  CountDecimate decimate = new CountDecimate(20);
51  Connector.connect(window, decimate);
52 
53  /* Prepare a plot */
54  Scatterplot plot = new Scatterplot();
55 
56  DrawPlot draw = new DrawPlot(plot);
57  Connector.connect(decimate, draw);
58 
59  /* Show plot in a JFrame */
60  BitmapJFrame frame = new BitmapJFrame();
61  Connector.connect(draw, frame);
62  frame.start();
63 
64  System.out.println("Displaying plot. Press Ctrl+C or close the window to end.");
65  while (true)
66  {
67  one.push();
68  Thread.sleep(10);
69  }
70  }
71 }
