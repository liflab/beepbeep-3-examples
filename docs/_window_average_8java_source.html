39 /**
40  * Compute the cumulative average of a list of numbers over a sliding window.
41  * Represented graphically, this example corresponds to the
42  * following chain of processors:
43  * <p>
44  * <img src="./doc-files/basic/WindowAverage.png" alt="Processor graph">
45  * @author Sylvain Hallé
46  * @difficulty Easy
47  */
48 public class WindowAverage
49 {
50  public static void main(String[] args)
51  {
52  /* Let us first create a source of arbitrary numbers. This is done
53  * by instantiating a {@link QueueSource} object, and giving to it
54  * an array of numbers. */
55  ///
56  QueueSource numbers = new QueueSource(1);
57  numbers.setEvents(new Object[]{2, 7, 1, 8, 2, 8, 1, 8, 2, 8,
58  4, 5, 9, 0, 4, 5, 2, 3, 5, 3, 6, 0, 2, 8, 7});
59 
60  /* We connect this source into a fork */
61  GroupProcessor group = new GroupProcessor(1, 1);
62  {
63  Fork fork = new Fork(2);
64  Cumulate sum_proc = new Cumulate(
65  new CumulativeFunction<Number>(Numbers.addition));
66  Connector.connect(fork, TOP, sum_proc, INPUT);
67 
68  /* Now we create a source of 1s and sum it; this is done with the same
69  * process as above, but on a stream that output the value 1 all the
70  * time. This effectively creates a counter outputting 1, 2, ... */
71  TurnInto ones = new TurnInto(1);
72  Connector.connect(fork, BOTTOM, ones, INPUT);
73  Cumulate counter = new Cumulate(
74  new CumulativeFunction<Number>(Numbers.addition));
75  Connector.connect(ones, OUTPUT, counter, INPUT);
76 
77  /* Divide one trace by the other; the output is the cumulative average
78  * of all numbers seen so far. */
79  ApplyFunction division = new ApplyFunction(Numbers.division);
80  Connector.connect(sum_proc, OUTPUT, division, LEFT);
81  Connector.connect(counter, OUTPUT, division, RIGHT);
82  group.addProcessors(fork, sum_proc, ones, counter, division);
83  group.associateInput(0, fork, 0);
84  group.associateOutput(0, division, 0);
85  }
86 
87  Window win = new Window(group, 3);
88  Connector.connect(numbers, win);
89  ///
90 
91  /* Extract the first 20 events from that pipe and print them. */
92  Pullable p = win.getPullableOutput();
93  System.out.println("The cumulative average is...");
94  for (int i = 0; i < 20; i++)
95  {
96  float value = (Float) p.pull();
97  System.out.print(value + ", ");
98  UtilityMethods.pause(1000);
99  }
100  System.out.println("done!");
101  }
102 }
