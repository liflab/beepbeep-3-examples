32 /**
33  * In this example, Machine A and Machine B are actually the same host; they
34  * just listen to different TCP ports on the same computer. Read this program
35  * first before looking that the other files in this package.
36  * <p>
37  * <img
38  * src="./doc-files/network/httppush/PushLocal.png"
39  * alt="Processor graph">
40  * @author Sylvain Hallé
41  *
42  */
43 public class PushLocalSerialize
44 {
45  public static void main(String[] args) throws ProcessorException, InterruptedException
46  {
47  ///
48  /* We first setup a {@link FunctionProcessor} that will execute
49  * the function {@link JsonSerializeString} on each input event. This function
50  * transforms an incoming object into a character string in the JSON format,
51  * through a process called <em>serialization</em>. Under the hood, the Azrael
52  * library takes care of this task. */
53  ApplyFunction serialize = new ApplyFunction(new JsonSerializeString());
54 
55  /* This function processor is connected to a {@link HttpUpstreamGateway}.
56  * The gateway is a processor that transmits its received events to another
57  * machine, through HTTP requests and responses. Since the chain operates in
58  * <em>push</em> mode, the gateway will be pushed character strings from
59  * upstream, and will in turn push them to the outside world by sending an HTTP
60  * request at a specific address. Thus, when instantiating the gateway, we must
61  * tell it the URL at which the request will be sent. In this case, the URL
62  * for Machine B is on the same host, on port 12144. The "/push" prefix is the
63  * "page" on Machine B the server will respond to. */
64  HttpUpstreamGateway up_gateway = new HttpUpstreamGateway("http://localhost:12144/push");
65 
66  /* We now move on to Machine B, which is responsible for receiving character
67  * strings and converting them back into objects. This is the mirror process of
68  * what was just done. So, the first step is to create an
69  * {@link HttpDownstreamGateway}. The gateway is instructed to listen to incoming
70  * requests on port 12144, to respond to requests made at the page "/push", and
71  * send through an HTTP <code>POST</code> request. */
72  HttpDownstreamGateway dn_gateway = new HttpDownstreamGateway(12144, "/push", Method.POST);
73 
74  /* Each event output from this processor is a sequence of character strings,
75  * taken from the payload of each HTTP request that is received. Remember
76  * that these strings are in the JSON format, and correspond to the serialized
77  * content of objects. The next step is to transform these strings back into
78  * objects, i.e. to <em>deserialize</em> them. This is done by a
79  * {@link FunctionProcessor} that applies the {@link JsonDeserializeString}
80  * function (again, using the Azrael library in the background). Note that
81  * this function must be given the class of the objects
82  * to be deserialized, so that it knows instances of what kind of objects
83  * to create. */
84  ApplyFunction deserialize = new ApplyFunction(
85  new JsonDeserializeString<CompoundObject>(CompoundObject.class));
86 
87  /* Just so that we can see something, we plug a {@link Print} processor at
88  * the end; it will print to the standard output whatever object it receives
89  * from upstream. */
90  Print print = new Print();
91 
92  /* We are now ready to pipe everything together. The interesting bit is
93  * what is <em>not</em> there: notice that we do not connect
94  * <code>up_gateway</code> and <code>dn_gateway</code>. Indeed, these two
95  * processors do not communicate using a BeepBeep pipe like the others;
96  * rather, <code>up_gateway</code> sends its events to <code>dn_gateway</code>
97  * through HTTP requests (i.e., outside of BeepBeep). This is what would
98  * make it possible to put the two halves of this processor chain
99  * (serialize and up_gateway on one side, dn_gateway, deserialize and
100  * print on the other) on two different machines. */
101  Connector.connect(serialize, up_gateway);
102  Connector.connect(dn_gateway, deserialize);
103  Connector.connect(deserialize, print);
104 
105  /* Since the gateways are actually mini-web servers, the servers need to
106  * be launched so that they can actually communicate. This is done by
107  * calling the {@link Processor#start() start()} method on both
108  * processors. Look out, as there can be only one server on a machine
109  * listening to a given TCP port; if you have another instance of this
110  * program already running, the call to start will throw an Exception. */
111  up_gateway.start();
112  dn_gateway.start();
113  ///
114 
115  /* We are now ready to push events and see what happens. First, we get
116  * a handle on the {@link Pushable} of the very first processor of the
117  * chain, <code>serialize</code> (which resides on Machine A). */
118  //!
119  Pushable p = serialize.getPushableInput();
120 
121  /* Let's push some dummy object. After the call to push, the standard
122  * output should display the contents of that object. This is pretty
123  * boring, but think about all the magic that happened in the background:
124  * <ul>
125  * <li>The object was converted into a JSON string</li>
126  * <li>The string was sent over the network through an HTTP request to
127  * another server...</li>
128  * <li>converted back into an object identical to the original...</li>
129  * <li>and pushed downstream to be handled by the rest of the
130  * processors as usual.</li>
131  * </ul>
132  * And so far, we've written about 10 lines of code. */
133  p.push(new CompoundObject(0, "foo", null));
134 
135  /* Sleep a little so you have time to look at the console... */
136  Thread.sleep(1000);
137 
138  /* Let's push again. You know the drill. */
139  p.push(new CompoundObject(0, "foo", new CompoundObject(6, "z", null)));
140  //!
141 
142  /* Once everything is done, we have to stop the servers to free the
143  * TCP ports on your machine. This is done by calling
144  * {@link Processor#stop() stop()} on both processors. */
145  up_gateway.stop();
146  dn_gateway.stop();
147 
148  /* That's all folks! */
149  ///
150  }
151 }
