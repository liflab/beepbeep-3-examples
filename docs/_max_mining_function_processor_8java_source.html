32 /**
33  * Create a mining function from BeepBeep processors. This example is
34  * identical to {@link MaxMiningFunction}, except that it uses BeepBeep
35  * processors to perform the same mining task, instead of custom code.
36  * As a result, the solution here is about 3 times shorter (in absolute
37  * number of lines of code, roughly 5 vs. 15).
38  * <p>
39  * The function is parameterized as follows:
40  * <p>
41  * <table>
42  * <tr><th>Parameter</th><th>Value</th></tr>
43  * <tr>
44  * <td><img src="./doc-files/mining/trenddistance/BetaProcessor.png" alt="Processor graph"></td>
45  * <td><img src="./doc-files/mining/extraction/MaxProcessor.png" alt="Processor graph"></td>
46  * </tr>
47  * <tr>
48  * <td><img src="./doc-files/mining/extraction/AlphaProcessor.png" alt="Processor graph"></td>
49  * <td><img src="./doc-files/mining/extraction/MaxOnSet.png" alt="Processor graph"></td>
50  * </tr>
51  * </table>
52  * <p>
53  * As one can see, the trend processor computes the maximum value on a stream.
54  * The aggregation processor runs yet another instance of the max processor,
55  * by feeding the elements of the input set one by one and returning its
56  * last value.
57  *
58  * @see MaxMiningFunction
59  * @author Sylvain Hall√©
60  */
61 public class MaxMiningFunctionProcessor
62 {
63  public static void main(String[] args) throws FunctionException, ConnectorException
64  {
65  /* First, we must get from somewhere a set of sequences. For the sake
66  * of this example, we just create a few dummy sequences of numbers
67  * from the contents of a file. */
68  Set<Sequence<Number>> sequences = SequenceReader.readNumericalSequences("numbers-1.csv");
69 
70  /* We then create an instance of our mining function. A mining function
71  * takes as input a set of sequences, and returns for its output some
72  * "pattern" extracted from this set of sequences. */
73  ProcessorMiningFunction<Number,Number> a_f = new ProcessorMiningFunction<Number,Number>(
74  new CumulativeProcessor(new CumulativeFunction<Number>(Maximum.instance)),
75  new ProcessOnSet(new CumulativeProcessor(new CumulativeFunction<Number>(Maximum.instance))));
76 
77  /* We then evaluate the function on our set of sequences. */
78  Number n = (Number) a_f.mine(sequences);
79 
80  /* In this case, the result of this function should be the maximum of
81  * all values in all the input sequences. */
82  System.out.println(n);
83  }
84 }
