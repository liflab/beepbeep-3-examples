24 /**
25  * Processor that computes the n-th Fibonacci number.
26  */
27 public class FibonacciProcessor extends UniformProcessor
28 {
29  public FibonacciProcessor()

31  super(1, 1);
32  }
33 
34  @Override
35  protected boolean compute(Object[] inputs, Object[] outputs)
36  {
37  // Perform some long computation
38  int index = (Integer) inputs[0];
39  BigInteger i = fib(index);
40  outputs[0] = i;
41  return true;
42  }

45  public FibonacciProcessor duplicate(boolean with_state)

50  /**
51  * Computes the n-th Fibonacci number.
52  * The actual result of this computation does not really matter;
53  * here we only use it as a CPU-intensive operation.
54  * @param nth The index of the number
55  * @return The Fibonacci number
56  */
57  static BigInteger fib(long nth)
58  {
59  nth = nth - 1;
60  long count = 0;
61  BigInteger first = BigInteger.ZERO;
62  BigInteger second = BigInteger.ONE;
63  BigInteger third = null;
64  while(count < nth)
65  {
66  third = new BigInteger(first.add(second).toString());
67  first = new BigInteger(second.toString());
68  second = new BigInteger(third.toString());
69  count++;
70  }
71  return third;
72  }
73 }
